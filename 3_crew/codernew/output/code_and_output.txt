# Python Code to Calculate the Series

def calculate_series(terms):
    total = 0
    for n in range(terms):
        term = (-1) ** n / (2 * n + 1)  # Calculate the nth term: 1/(2n+1) with alternating signs
        total += term
    return total * 4  # Multiply the sum by 4

# Calculate the series for the first 10,000 terms
result = calculate_series(10000)
print(result)
